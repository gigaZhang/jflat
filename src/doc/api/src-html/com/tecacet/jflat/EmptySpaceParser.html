001    package com.tecacet.jflat;
002    
003    import java.io.IOException;
004    import java.util.ArrayList;
005    import java.util.List;
006    import java.util.StringTokenizer;
007    
008    import com.tecacet.jflat.LineParser;
009    
010    /**
011     * Parses lines where the contents are seperated by anarbitrary number of spaces
012     * 
013     * @author Dimitri Papaioannou
014     *
015     */
016    public class EmptySpaceParser implements LineParser {
017    
018        @Override
019        public String[] parseLine(String line) throws IOException {
020            StringTokenizer st = new StringTokenizer(line, " ");
021            List<String> columns = new ArrayList<String>();
022            for (int col = 0; st.hasMoreTokens(); col++) {
023                String token = st.nextToken().trim();
024                columns.add(token);
025            } // end for each column
026            return columns.toArray(new String[columns.size()]);
027        }
028    
029    }
