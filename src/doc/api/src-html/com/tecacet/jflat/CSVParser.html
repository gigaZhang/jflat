016    package com.tecacet.jflat;
017    
018    import java.io.IOException;
019    import java.util.ArrayList;
020    import java.util.List;
021    
022    /**
023     * Parses a line into tokens based on a CSV separator. Uses a LineIterator to access the next line if an entry spans
024     * multiple lines. This is possible for values within quotes.
025     * 
026     * @author Dimitri Papaioannou
027     * 
028     */
029    public class CSVParser implements LineParser {
030    
031        /** The default separator to use if none is supplied to the constructor. */
032        private static final char DEFAULT_SEPARATOR = ',';
033    
034        /**
035         * The default quote character to use if none is supplied to the constructor.
036         */
037        private static final char DEFAULT_QUOTE_CHARACTER = '"';
038    
039        private char separator;
040    
041        private char quotechar;
042    
043        private LineIterator lineIterator;
044    
045        protected CSVParser() {
046    
047        }
048    
049        public CSVParser(LineIterator lineIterator) {
050            this.lineIterator = lineIterator;
051            this.separator = DEFAULT_SEPARATOR;
052            this.quotechar = DEFAULT_QUOTE_CHARACTER;
053        }
054    
055        public String[] parseLine(String nextLine) throws IOException {
056    
057            if (nextLine == null) {
058                return null;
059            }
060    
061            List<String> tokens = new ArrayList<String>();
062            StringBuffer sb = new StringBuffer();
063            boolean inQuotes = false;
064            do {
065                if (inQuotes) {
066                    // continuing a quoted section, re-append newline
067                    sb.append("\n");
068                    nextLine = lineIterator.getNextLine();
069                    if (nextLine == null)
070                        break;
071                }
072                for (int i = 0; i < nextLine.length(); i++) {
073    
074                    char c = nextLine.charAt(i);
075                    if (c == quotechar) {
076                        // this gets complex... the quote may end a quoted block, or
077                        // escape another quote.
078                        // do a 1-char lookahead:
079                        if (inQuotes // we are in quotes, therefore there can be
080                                // escaped quotes in here.
081                                && nextLine.length() > (i + 1) // there is indeed
082                                // another character
083                                // to check.
084                                && nextLine.charAt(i + 1) == quotechar) { // ..and
085                            // that char is a quote also.
086                            // we have two quote chars in a row == one quote char,
087                            // so consume them both and
088                            // put one on the token. we do *not* exit the quoted
089                            // text.
090                            sb.append(nextLine.charAt(i + 1));
091                            i++;
092                        } else {
093                            inQuotes = !inQuotes;
094                            // the tricky case of an embedded quote in the middle:
095                            // a,bc"d"ef,g
096                            if (i > 2 // not on the begining of the line
097                                    && nextLine.charAt(i - 1) != this.separator // not
098                                    // at the begining
099                                    // of an escape sequence
100                                    && nextLine.length() > (i + 1) && nextLine.charAt(i + 1) != this.separator // not
101                            // at the end of an escape sequence
102                            ) {
103                                sb.append(c);
104                            }
105                        }
106                    } else if (c == separator && !inQuotes) {
107                            // an empty field equates to null
108                        tokens.add(sb.length() > 0 ? sb.toString() : null);
109                        
110                        // start work on next token, if there is one
111                        sb = new StringBuffer(); 
112                    } else {
113                        sb.append(c);
114                    }
115                }
116            } while (inQuotes);
117            // to account for final field...
118            tokens.add(sb.length() > 0 ? sb.toString() : null);
119            return tokens.toArray(new String[0]);
120    
121        }
122    
123        public char getSeparator() {
124            return separator;
125        }
126    
127        public void setSeparator(char separator) {
128            this.separator = separator;
129        }
130    
131        public char getQuotechar() {
132            return quotechar;
133        }
134    
135        public void setQuotechar(char quotechar) {
136            this.quotechar = quotechar;
137        }
138    
139        public LineIterator getLineIterator() {
140            return lineIterator;
141        }
142    
143        public void setLineIterator(LineIterator lineIterator) {
144            this.lineIterator = lineIterator;
145        }
146    
147    }
