001    package com.tecacet.jflat;
002    
003    /**
004     * 
005     * @author Dimitri Papaioannou
006     *
007     * This interface is used by a File Writer to extract a column value from a bean.
008     * Suppose that you want to map a currency ($) and an amount (10,000) to a single value that looks like this: $10,000
009     * Then you can use a value extractor like this:
010     * 
011     * <verbatim>
012     *  String getValue(Order order) { return order.getCurrency().getSymbol() + order.getAmount() }
013     * </verbatim>
014     * 
015     */
016    public interface ValueExtractor<T> {
017    
018        String getValue(T bean);
019    }
